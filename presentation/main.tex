% ---------------- 4 · ERA 1 FAILURES ----------------------------------------------
\begin{frame}{Era 1: weekend GRPO, everything at 14.3\%}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \footnotesize
      \begin{enumerate}\itemsep0.4em
        \item Metal OOM: group\,6 / completion\,512 blew 48\,GB at iter 2; survived at group\,4 / comp\,256 ($\sim$38\,GB peak).
        \item $\sigma=0$ trap: at 0\% base pass every rollout scores the same, so advantage $(r-\bar r)/\sigma = 0$, KL $=0.0$ forever. \alert{RL cannot bootstrap a skill the base has none of.}
        \item Splice bug: models emit the whole \texttt{can \ldots\ \{\}} unit; the hole sat \emph{inside} it, so the result was nested broken Jac. Base, warm (SFT loss 0.006!), and GRPO scored identically: the splice was broken, not the models.
        \item Dense wall: Qwen3.6-27B SFT OOM'd at iter 1 in every config; the whole line is inference-only on 48\,GB.
        \item LoRA-GRPO null: splice fixed, real reward variance ($\sigma$ up to 0.11), and +GRPO output was still \alert{byte-identical} to its start point, KL${}\approx{}$0.
      \end{enumerate}
    \end{column}
    \begin{column}{0.41\textwidth}
      \small\centering
      \begin{tabular}{@{}lc@{}}
        \toprule
        Phase 1 (31 tasks, holdout 7) & pass \\
        \midrule
        jac-qwen3coder base & 14.3\% \\
        \quad + GRPO & 14.3\% \\
        qwen3coder base & 0\% \\
        \quad + warm-start SFT & 14.3\% \\
        \quad + GRPO & 14.3\% \\
        \bottomrule
      \end{tabular}

      \vspace{3mm}
      \scriptsize\color{inkmuted}
      Phase 2.5 (51 tasks, holdout 12): pass@1 $=$ pass@8 $=$ 14.3\% for every arm.
      STaR flickered to pass@4 $=$ 25\% for one round, then died.
    \end{column}

\end{frame}

% ---------------- 6 · ERA 2: LADDER + FALSE VERDICT -------------------------------
\begin{frame}{Era 2: the SFT ladder, built to settle it}
  \small
  \begin{itemize}\itemsep0.6em
    \item Design (leak-free rebuild): rungs train-$N \in \{1,3,5,10,20,\text{all}\}$; conditions base, SFT, SFT+GRPO, raw-GRPO control, tuned-GRPO (500 iters, 10$\times$ LR).
    \item 2 models $\times$ 30 cells; corpus grown 51$\rightarrow$84 deterministic tasks.
    \item Sound train/holdout splits, monotone reward, Wilson CIs throughout, built to be trustworthy this time.
  \end{itemize}
  \vspace{4mm}
  \centering
  \setbeamercolor{block title}{bg=jaseciorange,fg=white}
  \setbeamercolor{block body}{bg=jaseciorange!8,fg=ink}
  \begin{block}{Why a flat result is a strong prior something's wrong}
    \small A real signal usually varies at least a little between conditions. A
    number that's \emph{exactly} flat across 30 cells of real training runs says
    you're not measuring the model. You're measuring a fixed artifact of the
    harness. That's what the next slide shows, and at the time it read as proof.
  \end{block}
\end{frame}

% ---------------- 8 · THE MEASUREMENT BUG -----------------------------------------
\begin{frame}{The measurement bug that faked the dead end}
  \small
  \begin{itemize}\itemsep0.6em
    \item \texttt{extract\_jac}/\texttt{unwrap\_unit} grabbed the \alert{driver docstring} instead of the model's answer whenever the model echoed the whole driver, so it auto-failed.
    \item The GRPO reward used the \alert{same extractor}, so GRPO was trained \emph{and} scored on garbage. Era 1's real LoRA-GRPO null and this eval bug were two separate problems stacked on top of each other.
    \item Fixed in \texttt{8164ee2}: brace-matched, name-targeted body extraction, then everything re-run.
  \end{itemize}
  \vspace{3mm}
  \setbeamercolor{block title}{bg=jasecidark,fg=white}
  \setbeamercolor{block body}{bg=jasecidark!8,fg=ink}
  \begin{block}{Lesson}
    \small Verify the grader before trusting a null result. One shared helper silently capped three weeks of experiments.
  \end{block}
\end{frame}

% ---------------- 14 · WHY GRPO ~= SFT --------------------------------------------
\begin{frame}{Why GRPO $\approx$ SFT, and why that's a real result}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \small
      \begin{itemize}\itemsep0.6em
        \item Every measurement here agrees: GRPO on top of SFT adds nothing beyond SFT alone; raw GRPO from the base moves nothing. This is Era 1's LoRA-GRPO null (KL$\approx$0, byte-identical output) showing up again at scale.
        \item Matches Yue et al.\ 2504.13837: RL sharpens a model's \emph{sampling distribution} around capability it already has. It doesn't expand the underlying boundary.
        \item Once SFT already moves greedy close to the model's own pass@$k$ ceiling, there's little room left to sharpen further, and LoRA's limited capacity (same constraint behind task interference) leaves even less room to search past it.
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \setbeamercolor{block title}{bg=jaseciorange,fg=white}
      \setbeamercolor{block body}{bg=jaseciorange!8,fg=ink}
      \begin{block}{What would actually expand it}
        \small
        \begin{itemize}\itemsep0.4em
          \item Full fine-tune: removes LoRA's capacity ceiling entirely. Untested here; needs $>$48\,GB.
          \item Distillation / expert iteration: train on a stronger teacher's, or the model's own compiler-verified best, outputs. Injects capability from \emph{outside} rather than re-weighting what's already there.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ---------------- 19 · GLOSSARY ----------------------------------------------------
\begin{frame}[t]{Glossary: terms \& context}
  \tiny
  \begin{columns}[T]
    \begin{column}{0.32\textwidth}
      \begin{itemize}\itemsep0.15em
        \item \term{Jac}: Jaseci's language; supersets Python, adds graph-native OSP constructs.
        \item \term{OSP / walkers}: \texttt{node}/\texttt{edge}/\texttt{walker} idiom; no Python equivalent.
        \item \term{SFT}: supervised fine-tuning on prompt$\rightarrow$answer pairs.
        \item \term{DPO}: preference optimization on chosen-vs-rejected pairs.
        \item \term{GRPO}: RL, sample rollout groups, advantage $=(r-\bar r)/\sigma$.
        \item \term{tuned-GRPO}: GRPO variant, 500 iters, 10$\times$ LR; rules out ``just needs more training.''
        \item \term{LoRA}: low-rank adapters; only small matrices train.
        \item \term{MoE / 30B-A3B}: 30B params, $\sim$3B active/token.
        \item \term{MLX}: Apple's on-device ML framework; runs the LoRA training here.
        \item \term{Metal}: Apple's GPU compute API; where the OOMs actually happen.
      \end{itemize}
    \end{column}
    \begin{column}{0.32\textwidth}
      \begin{itemize}\itemsep0.15em
        \item \term{unified memory}: Apple silicon's shared CPU/GPU RAM; 48\,GB hard ceiling here.
        \item \term{warm-start}: brief SFT before RL so rollouts aren't all-fail.
        \item \term{STaR}: sample, keep verified wins, SFT on them, repeat.
        \item \term{$\sigma=0$ trap}: equal rollout rewards, so zero gradient; can't bootstrap from 0\% skill.
        \item \term{KL}: divergence from start policy; $\approx$0 means the model didn't move.
        \item \term{this\_is\_jac}: real Jac codebase mined for hole-fill tasks; ground-truth source.
        \item \term{exact-stdout}: grading rule, output must byte-match; no partial credit.
        \item \term{Studio}: this repo's app; ships the GENERATE panel + RL data pipeline live.
        \item \term{holdout}: eval tasks never trained on; five sets ($n{=}18/17/11/16/32$).
      \end{itemize}
    \end{column}
    \begin{column}{0.32\textwidth}
      \begin{itemize}\itemsep0.15em
        \item \term{hole-fill}: one function body blanked from a working Jac file.
        \item \term{conversion}: python$\rightarrow$jac translation; richest spec, best scores.
        \item \term{ladder / rung-$N$}: train on $N$ tasks, eval on holdout.
        \item \term{task interference}: harder train tasks regress ones already learned.
        \item \term{greedy / pass@1}: one deterministic (temp-0) answer; the headline metric.
        \item \term{pass@$k$ (oracle)}: any of $k$ samples correct; ceiling if perfectly picked.
        \item \term{best-of-$k$ (deploy)}: sample $k$, return first the compiler accepts.
        \item \term{Wilson CI}: binomial confidence interval; the noise bar on every rate.
        \item \term{expert iteration / distillation}: train on a stronger model's verified outputs.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
