% ---------------- 4 · ERA 0 FAILURES ----------------------------------------------
\begin{frame}{Era 0 --- weekend GRPO: everything at 14.3\%}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \footnotesize
      \begin{enumerate}\itemsep0.4em
        \item \textbf{Metal OOM} --- group\,6 / completion\,512 blew 48\,GB at iter 2; survived at group\,4 / comp\,256 ($\sim$38\,GB peak).
        \item \textbf{$\sigma=0$ trap} --- at 0\% base pass every rollout scores the same $\Rightarrow$ advantage $(r-\bar r)/\sigma = 0$, KL $=0.0$ forever. \alert{RL cannot bootstrap a skill the base has none of.}
        \item \textbf{Splice bug} --- models emit the whole \texttt{can \ldots\ \{\}} unit; the hole sat \emph{inside} it $\Rightarrow$ nested broken Jac. Base, warm (SFT loss 0.006!), and GRPO scored identically --- the splice was broken, not the models.
        \item \textbf{Dense wall} --- Qwen3.6-27B SFT OOM'd at iter 1 in every config; the whole line is inference-only on 48\,GB.
        \item \textbf{LoRA-GRPO null} --- splice fixed, real reward variance ($\sigma$ up to 0.11): +GRPO output \alert{byte-identical} to its start point, KL${}\approx{}$0.
      \end{enumerate}

% ---------------- 15 · GLOSSARY ----------------------------------------------------
\begin{frame}[t]{Glossary --- terms \& context}
  \scriptsize
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{itemize}\itemsep0.25em
        \item \term{Jac} --- Jaseci's language; supersets Python, adds graph-native OSP constructs.
        \item \term{OSP / graph walkers} --- Object-Spatial Programming: \texttt{node}/\texttt{edge}/\texttt{walker} idiom; no Python equivalent.
        \item \term{SFT} --- supervised fine-tuning on prompt$\rightarrow$answer pairs.
        \item \term{DPO} --- direct preference optimization; trains on chosen-vs-rejected pairs.
        \item \term{GRPO} --- group-relative policy optimization (RL): sample rollout groups, advantage $=(r-\bar r)/\sigma$.
        \item \term{LoRA} --- low-rank adapters; only small matrices train. Whole pipeline: LoRA on MLX (Apple-silicon ML).
        \item \term{MoE / 30B-A3B} --- mixture-of-experts, 30B params, $\sim$3B active per token --- why it trains on 48\,GB.
        \item \term{warm-start} --- brief SFT before RL so rollouts aren't all-fail.
        \item \term{STaR} --- self-taught reasoner loop: sample, keep verified wins, SFT on them, repeat.
        \item \term{$\sigma=0$ trap} --- all rollouts score equal $\Rightarrow$ advantage 0 $\Rightarrow$ zero gradient; RL can't start from 0\% skill.
        \item \term{KL} --- divergence from start policy during RL; $\approx$0 $=$ model didn't move.
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{itemize}\itemsep0.25em
        \item \term{greedy / pass@1} --- one deterministic (temp-0) answer; the single-shot metric.
        \item \term{pass@$k$ (oracle)} --- any of $k$ samples correct; ceiling if a perfect verifier picked.
        \item \term{best-of-$k$ (deploy)} --- sample $k$, return first the compiler accepts; $=$ oracle here.
        \item \term{holdout} --- eval tasks never trained on; five sets ($n=18/17/11/16/32$).
        \item \term{hole-fill} --- task format: one function body blanked out of a working Jac file; exact-stdout grading.
        \item \term{conversion} --- python$\rightarrow$jac translation tasks; richest spec, best scores.
        \item \term{ladder / rung-$N$} --- train on $N$ tasks, eval on holdout; rungs 1$\rightarrow$all isolate data-size effects.
        \item \term{task interference} --- adding harder train tasks regresses learned ones (rung-20 $\rightarrow$ all: 61$\rightarrow$56\%).
        \item \term{Wilson CI} --- binomial confidence interval; the noise bar on every pass rate.
        \item \term{expert iteration / distillation} --- train on a stronger model's verified outputs; expands ability.
      \end{itemize}
    \end{column}
  \end{columns}
